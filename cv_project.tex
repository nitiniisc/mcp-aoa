\resumeProjectHeading
  {\textbf{MCP-based Agent of Agents (AoA)} $|$
   \emph{Claude Agent SDK, MCP, Python, LLM Orchestration}}
  {2026}
\resumeItemListStart
  \resumeItem{
    Designed and implemented an \textbf{Agent of Agents (AoA)} architecture
    using the \textbf{Claude Agent SDK} (claude-opus-4-6), where an
    \textbf{orchestrator LLM} dynamically delegates tasks to specialist
    subagents — researcher, calculator, and summarizer.
  }
  \resumeItem{
    Exposed custom domain tools (\texttt{search\_knowledge},
    \texttt{run\_calculation}, \texttt{summarize\_text}) via an in-process
    \textbf{MCP (Model Context Protocol)} server, enabling modular
    \textbf{tool use} and clean separation of agent capabilities.
  }
  \resumeItem{
    Integrated \textbf{real-time web search} (WebSearch tool) into the
    researcher subagent as a fallback, combining retrieval-augmented
    generation (RAG) with live internet data for up-to-date responses.
  }
  \resumeItem{
    Implemented \textbf{streaming token output} via
    \texttt{ClaudeSDKClient.receive\_response()}, supporting long-horizon
    agentic tasks with \textbf{multi-turn tool-use loops} and configurable
    \texttt{max\_turns} safety limits.
  }
  \resumeItem{
    Hosted on GitHub (\href{https://github.com/nitiniisc/mcp-aoa}
    {\texttt{nitiniisc/mcp-aoa}});
    extensible design allows adding new subagents or connecting external
    MCP servers (e.g., PostgreSQL, Playwright) with minimal code changes.
  }
\resumeItemListEnd
